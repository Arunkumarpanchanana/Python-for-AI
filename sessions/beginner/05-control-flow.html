Control Flow & Loops

Control what runs and when — loops and conditions power every AI pipeline.

25 min Beginner Hands-on Lab

if / elif / else

score = 85
if score >= 90:   grade = "A"
elif score >= 80: grade = "B"
elif score >= 70: grade = "C"
else:             grade = "F"
print(grade)   # B

label = "pass" if score >= 70 else "fail"

for Loops

for fruit in ["apple","banana","cherry"]:
    print(fruit)

for i in range(1,10,2):   # 1,3,5,7,9
    print(i)

for i, val in enumerate(["a","b","c"]):
    print(i, val)

for name, sc in zip(["Alice","Bob"],[90,85]):
    print(f"{name}: {sc}")

while + break / continue

count = 0
while count < 5:
    count += 1
    print(count)

for n in range(10):
    if n == 3: continue  # skip
    if n == 7: break     # stop
    print(n)

Dict Comprehensions

data = {"a":1,"b":2,"c":3}
squared = {k: v**2 for k,v in data.items()}
print(squared)  # {"a":1,"b":4,"c":9}

FizzBuzz

    if n%15==0:   print("FizzBuzz")
    elif n%3==0:  print("Fizz")
    elif n%5==0:  print("Buzz")

Sum of evens

total = sum(x for x in range(1,101) if x%2==0)
print(f"Sum of evens 1-100: {total}")  # 2550
3

Guessing game simulation

import random
secret = random.randint(1,10)
for guess in [3, 7, secret]:
    if guess == secret: print(f"Correct! {secret}"); break
    elif guess < secret: print(f"{guess} too low")
    else: print(f"{guess} too high")
