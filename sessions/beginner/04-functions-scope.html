Functions & Scope

Write reusable, clean functions — the core skill for AI application development.

30 min Beginner Hands-on Lab

Defining Functions

def greet(name):
    return f"Hello, {name}!"

def score(correct, total, bonus=0):
    return min((correct/total)*100 + bonus, 100)

print(score(8, 10, 5))   # 85.0

*args and **kwargs

def add_all(*args):
    return sum(args)
print(add_all(1,2,3,4))   # 10

def build_prompt(**kwargs):
    return "\n".join(f"{k}: {v}" for k,v in kwargs.items())
print(build_prompt(role="assistant", tone="friendly"))

Scope — LEGB Rule

x = "global"
def outer():
    x = "outer"
    def inner():
        x = "inner"; print(x)  # "inner"
    inner()
    print(x)   # "outer"
outer()
print(x)   # "global"

Lambda Functions

square = lambda x: x**2

students = [{"name":"Bob","grade":85},{"name":"Alice","grade":92}]
students.sort(key=lambda s: s["grade"], reverse=True)
print(students[0]["name"])   # Alice

Type Hints and Docstrings

def count_tokens(text: str, chars_per_token: float = 4.0) -> int:
    """Estimate token count for an LLM prompt."""
    return int(len(text) / chars_per_token)

print(count_tokens("Hello world"))  # 2

Lab: Text Analyzer

Build reusable text analysis functions.

1

Word count and avg length

def word_count(text): return len(text.split())
def avg_word_length(text):
    words = text.split()
    return sum(len(w) for w in words)/len(words) if words else 0

text = "Python is the best language for AI"
print(word_count(text), avg_word_length(text))
2

Most common word

from collections import Counter
def top_word(text):
    return Counter(text.lower().split()).most_common(1)[0][0]
print(top_word("the cat sat on the mat and the cat"))
3

Combined analyzer

def analyze(text):
    return {"words":word_count(text),"top":top_word(text),"chars":len(text)}
for k,v in analyze(text).items(): print(f"{k}: {v}")
