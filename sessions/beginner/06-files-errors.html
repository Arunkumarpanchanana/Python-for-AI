File Handling & Error Handling

Read, write, and process files — essential for loading datasets and saving AI results.

30 min Beginner Hands-on Lab

Reading Files

with open("data.txt", "r") as f:
    content = f.read()      # whole file

with open("data.txt", "r") as f:
    lines = f.readlines()   # list of lines

with open("data.txt", "r") as f:
    for line in f:          # memory-efficient
        print(line.strip())

Writing Files

with open("out.txt", "w") as f:   # creates/overwrites
    f.write("Hello!\n")

with open("log.txt", "a") as f:   # appends
    f.write("New entry\n")

Error Handling

try:
    value = int("not a number")
except ValueError as e:
    print(f"Error: {e}")
except FileNotFoundError:
    print("File missing")
finally:
    print("Always runs")

def get_score(n):
    if not 0 <= n <= 100:
        raise ValueError(f"Score {n} out of range")
    return n

Lab: CSV to JSON

Read sales CSV, aggregate, save as JSON.

1

Create CSV

import csv, json
rows=[["product","amount"],["Widget","150"],["Gadget","200"],["Widget","75"]]
with open("sales.csv","w",newline="") as f: csv.writer(f).writerows(rows)
2

Aggregate

totals = {}
with open("sales.csv") as f:
    for row in csv.DictReader(f):
        p = row["product"]
        totals[p] = totals.get(p,0) + int(row["amount"])
print(totals)
3

Save JSON

try:
    with open("summary.json","w") as f: json.dump(totals,f,indent=2)
    print("Saved!")
except IOError as e: print(f"Error: {e}")
