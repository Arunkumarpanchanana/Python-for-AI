Object-Oriented Programming

Build reusable, organised code with classes — the pattern behind every major AI library.

35 min Beginner Hands-on Lab

Classes and Objects

class Dog:
    species = "Canis familiaris"
    def __init__(self, name, breed):
        self.name  = name
        self.breed = breed
    def speak(self):
        return f"{self.name} says Woof!"
    def __str__(self):
        return f"Dog({self.name}, {self.breed})"

rex = Dog("Rex", "Labrador")
print(rex.speak())
print(str(rex))

Inheritance

class Animal:
    def __init__(self, name): self.name = name
    def speak(self): return f"{self.name} makes a sound"

class Cat(Animal):
    def speak(self): return f"{self.name} says Meow!"

class Duck(Animal):
    def speak(self): return f"{self.name} says Quack!"

for a in [Cat("Whiskers"), Duck("Donald")]:
    print(a.speak())

Magic Methods

class Vector:
    def __init__(self,x,y): self.x=x; self.y=y
    def __repr__(self): return f"Vector({self.x},{self.y})"
    def __add__(self,o): return Vector(self.x+o.x,self.y+o.y)
    def __len__(self): return 2

v1=Vector(1,2); v2=Vector(3,4)
print(v1+v2)   # Vector(4,6)
When to use OOP: data + behaviour belong together, you need multiple instances, or you're building something others will extend.

Lab: ChatHistory Class

Build an OOP conversation manager — core of any chatbot.

1

Define the class

import json
class ChatHistory:
    def __init__(self, system="You are a helpful assistant"):
        self.system = system; self.messages = []
    def add(self, role, content):
        self.messages.append({"role":role,"content":content})
    def get_all(self):
        return [{"role":"system","content":self.system}] + self.messages
    def clear(self): self.messages = []
    def __len__(self): return len(self.messages)
    def __str__(self): return f"ChatHistory({len(self)} messages)"
2

Use it

chat = ChatHistory("You are a Python tutor.")
chat.add("user", "What is a list?")
chat.add("assistant", "A list is an ordered collection...")
print(chat)               # ChatHistory(2 messages)
print(chat.get_all()[0])  # system prompt
3

Save to JSON

with open("chat.json","w") as f:
    json.dump(chat.get_all(), f, indent=2)
print("Saved!")
