Lab: Daily Motivation Tool

Use os, datetime, random, and Counter to build a useful script.

1

Build the tool

import random, os
from datetime import datetime
from collections import Counter

quotes = [
    "First solve the problem, then write the code.",
    "Code is like humor: when you explain it, its bad.",
    "Make it work, make it right, make it fast.",
]
random.seed(datetime.now().strftime("%Y-%m-%d"))
user = os.environ.get("USER","Learner")
print(f"Good morning, {user}!")
print(f"Quote: {random.choice(quotes)}")

Word frequency

all_words = " ".join(quotes).lower().split()
freq = Counter(all_words)
for word, count in freq.most_common(5):
    print(f"  {word}: {count}")
