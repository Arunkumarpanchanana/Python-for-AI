What Is an AI Data Pipeline?

An AI data pipeline is a series of automated steps that takes raw data → cleans it → enriches it with AI → stores the results. Instead of running scripts manually, the pipeline runs on a schedule or trigger and processes data at scale.

Example: Fetch customer reviews → clean text → classify sentiment with AI → extract key topics → store in database → send weekly report. This entire workflow can be automated in Python.

Pipeline Architecture

from dataclasses import dataclass
from typing import Optional

@dataclass
class Review:
    id: int
    text: str
    source: str
    sentiment: Optional[str] = None
    topics: Optional[list] = None
    processed: bool = False

# Each stage is a pure function: data in, data out
def fetch_reviews(source_url: str) -> list[Review]:   ...
def clean_reviews(reviews: list[Review]) -> list[Review]:  ...
def classify_sentiment(reviews: list[Review]) -> list[Review]:  ...
def extract_topics(reviews: list[Review]) -> list[Review]:  ...
def save_results(reviews: list[Review], db_path: str) -> int:  ...

def run_pipeline(source_url: str, db_path: str):
    print("Fetching...")
    reviews = fetch_reviews(source_url)
    print(f"Cleaning {len(reviews)} reviews...")
    reviews = clean_reviews(reviews)
    print("Classifying sentiment...")
    reviews = classify_sentiment(reviews)
    print("Extracting topics...")
    reviews = extract_topics(reviews)
    n = save_results(reviews, db_path)
    print(f"Pipeline done. Saved {n} reviews.")

Batch AI Processing with Rate Limiting

import time
from anthropic import Anthropic

client = Anthropic()

def classify_batch(reviews: list[Review], batch_size=10) -> list[Review]:
    results = []
    for i in range(0, len(reviews), batch_size):
        batch = reviews[i:i+batch_size]
        for review in batch:
            try:
                r = client.messages.create(
                    model="claude-opus-4-5", max_tokens=10,
                    messages=[{"role":"user","content":
                        f"Classify as positive/negative/neutral:\n{review.text}\nRespond with one word only."
                    }]
                )
                review.sentiment = r.content[0].text.strip().lower()
            except Exception as e:
                print(f"Error on review {review.id}: {e}")
                review.sentiment = "error"
            results.append(review)
        time.sleep(0.5)  # rate limit: 2 batches/sec max
        print(f"  Processed {min(i+batch_size, len(reviews))}/{len(reviews)}")
    return results

Saving to SQLite (Simple Database)

import sqlite3, json

def save_to_db(reviews: list[Review], db="reviews.db") -> int:
    conn = sqlite3.connect(db)
    conn.execute("""CREATE TABLE IF NOT EXISTS reviews (
        id INTEGER PRIMARY KEY, text TEXT, source TEXT,
        sentiment TEXT, topics TEXT, processed INTEGER DEFAULT 1
    )""")
    rows = [(r.id, r.text, r.source, r.sentiment, json.dumps(r.topics), 1) for r in reviews]
    conn.executemany("INSERT OR REPLACE INTO reviews VALUES (?,?,?,?,?,?)", rows)
    conn.commit()
    n = len(rows)
    conn.close()
    return n

def query_results(db="reviews.db") -> None:
    conn = sqlite3.connect(db)
    for row in conn.execute("SELECT sentiment, COUNT(*) FROM reviews GROUP BY sentiment"):
        print(f"{row[0]:10s}: {row[1]}")
    conn.close()
Scaling up: When your pipeline needs to run faster, switch from sequential processing to asyncio.gather() for concurrent API calls. For very large datasets (millions of rows), look at Apache Airflow or Prefect for orchestration.

2

Clean and classify with AI

import anthropic, re, time, os
client = anthropic.Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY",""))

def clean(text):
    return re.sub(r'\s+', ' ', text).strip()

def classify(text):
    r = client.messages.create(
        model="claude-opus-4-5", max_tokens=5,
        messages=[{"role":"user","content":
            f"Classify as: positive, negative, or neutral. One word only.\n{text}"}]
    )
    return r.content[0].text.strip().lower()

results = []
for rev in reviews_raw:
    cleaned = clean(rev["text"])
    sentiment = classify(cleaned)
    results.append({**rev, "cleaned": cleaned, "sentiment": sentiment})
    print(f"  [{rev['id']}] {sentiment:9s} | {cleaned[:50]}")
    time.sleep(0.3)
3

Save to database and query

import sqlite3, json

conn = sqlite3.connect("reviews.db")
conn.execute("CREATE TABLE IF NOT EXISTS reviews (id INTEGER PRIMARY KEY, text TEXT, cleaned TEXT, sentiment TEXT, source TEXT)")
for r in results:
    conn.execute("INSERT OR REPLACE INTO reviews VALUES (?,?,?,?,?)",
        (r["id"], r["text"], r["cleaned"], r["sentiment"], r["source"]))
conn.commit()

print("\nSentiment breakdown:")
for row in conn.execute("SELECT sentiment, COUNT(*) n FROM reviews GROUP BY sentiment ORDER BY n DESC"):
    print(f"  {row[0]:10s}: {row[1]}")
conn.close()
