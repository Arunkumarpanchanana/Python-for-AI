Why Wrap AI in an API?

A Python script is a local tool. A Flask API is a service — it can be called by web apps, mobile apps, Slack bots, or any other client. Wrapping your AI in a REST API is the standard way to productionize it.

pip install flask anthropic

Your First Flask AI Endpoint

from flask import Flask, request, jsonify
from anthropic import Anthropic
import os

app = Flask(__name__)
client = Anthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

@app.route("/chat", methods=["POST"])
def chat():
    data = request.get_json()
    if not data or "message" not in data:
        return jsonify({"error": "message required"}), 400
    
    response = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=512,
        messages=[{"role":"user","content":data["message"]}]
    )
    return jsonify({"reply": response.content[0].text})

if __name__ == "__main__":
    app.run(debug=True, port=5000)

Test it: curl -X POST http://localhost:5000/chat -H "Content-Type: application/json" -d '{"message":"What is Python?"}'

Adding Input Validation and Error Handling

from functools import wraps
import anthropic

def require_json(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        if not request.is_json:
            return jsonify({"error": "Content-Type must be application/json"}), 415
        return f(*args, **kwargs)
    return decorated

@app.route("/summarise", methods=["POST"])
@require_json
def summarise():
    data = request.get_json()
    text = data.get("text", "").strip()
    if not text:
        return jsonify({"error": "text is required"}), 400
    if len(text) > 50_000:
        return jsonify({"error": "text too long (max 50,000 chars)"}), 400
    try:
        response = client.messages.create(
            model="claude-opus-4-5", max_tokens=300,
            messages=[{"role":"user","content":f"Summarise in 3 bullet points:\n\n{text}"}]
        )
        return jsonify({"summary": response.content[0].text})
    except anthropic.APIError as e:
        return jsonify({"error": str(e)}), 502

Multiple Endpoints and a Health Check

@app.route("/health")
def health():
    return jsonify({"status": "ok", "model": "claude-opus-4-5"})

@app.route("/classify", methods=["POST"])
def classify():
    data = request.get_json()
    text  = data.get("text", "")
    cats  = data.get("categories", ["positive", "negative", "neutral"])
    prompt = f"Classify the following text into exactly one of: {cats}.\nRespond with only the category name.\n\nText: {text}"
    response = client.messages.create(
        model="claude-opus-4-5", max_tokens=20,
        messages=[{"role":"user","content":prompt}]
    )
    label = response.content[0].text.strip().lower()
    return jsonify({"label": label, "text": text})

Lab: Build a 3-Endpoint AI API

Build a Flask API with chat, summarise, and classify endpoints — then test it with curl.

1

Create the Flask app file

In Colab, create a file called app.py:

%%writefile app.py
from flask import Flask, request, jsonify
from anthropic import Anthropic
import os

app = Flask(__name__)
client = Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY",""))

@app.route("/health")
def health():
    return jsonify({"status":"ok"})

@app.route("/chat", methods=["POST"])
def chat():
    msg = request.json.get("message","")
    r = client.messages.create(model="claude-opus-4-5",max_tokens=200,
        messages=[{"role":"user","content":msg}])
    return jsonify({"reply": r.content[0].text})

@app.route("/summarise", methods=["POST"])
def summarise():
    text = request.json.get("text","")
    r = client.messages.create(model="claude-opus-4-5",max_tokens=200,
        messages=[{"role":"user","content":f"Summarise in 2 sentences: {text}"}])
    return jsonify({"summary": r.content[0].text})

if __name__ == "__main__":
    app.run(port=5000)
