Build your core feature

Start with the single most important feature of your chosen track. Get it working end-to-end before adding anything else. Refer back to the session notebooks for copy-paste patterns.

# Track A starter: Writing Coach
import streamlit as st
from anthropic import Anthropic
import os

client = Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY",""))
text = st.text_area("Paste your text:")
if st.button("Improve") and text:
    r = client.messages.create(
        model="claude-opus-4-5", max_tokens=500,
        messages=[{"role":"user","content":f"Make this more professional:\n{text}"}])
    st.write(r.content[0].text)
3
