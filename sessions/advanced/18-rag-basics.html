What Is RAG and Why Does It Matter?

Retrieval-Augmented Generation (RAG) solves one of the biggest LLM limitations: models only know what was in their training data. RAG lets you give any LLM access to your own documents — PDFs, wikis, databases — at query time, without fine-tuning.

The pipeline: Chunk documents → Embed them into vectors → Store in a vector database → At query time: Embed the question → Retrieve similar chunks → Generate answer with the retrieved context.

Why it works: Embedding converts text into a list of numbers where semantically similar texts have similar numbers (close in vector space). Searching by similarity finds relevant chunks even if the exact words differ.

Step 1: Prepare Documents

def load_and_chunk(filepath: str, chunk_size=500, overlap=50) -> list[dict]:
    with open(filepath) as f:
        text = f.read()
    chunks = []
    start = 0
    while start < len(text):
        end = start + chunk_size
        chunks.append({
            "id": len(chunks),
            "text": text[start:end],
            "source": filepath
        })
        start += chunk_size - overlap
    return chunks

chunks = load_and_chunk("my_document.txt")
print(f"Created {len(chunks)} chunks")

Step 2: Embed Chunks

from openai import OpenAI
import numpy as np

client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

def embed(texts: list[str]) -> np.ndarray:
    """Embed a list of texts, returns (n, 1536) array."""
    response = client.embeddings.create(
        model="text-embedding-3-small",
        input=texts
    )
    vectors = [r.embedding for r in response.data]
    return np.array(vectors)

# Embed all chunks
texts = [c["text"] for c in chunks]
vectors = embed(texts)
print("Embedding shape:", vectors.shape)  # (n_chunks, 1536)

Step 3: Retrieve Relevant Chunks

def cosine_similarity(a: np.ndarray, b: np.ndarray) -> float:
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

def retrieve(query: str, vectors: np.ndarray, chunks: list, top_k=3) -> list[dict]:
    query_vec = embed([query])[0]
    scores = [cosine_similarity(query_vec, v) for v in vectors]
    top_indices = np.argsort(scores)[::-1][:top_k]
    return [chunks[i] for i in top_indices]

results = retrieve("What is the main topic?", vectors, chunks)
for r in results:
    print(f"Score chunk {r['id']}: {r['text'][:80]}...")

Step 4: Generate Answer with Context

from anthropic import Anthropic
ac = Anthropic()

def rag_answer(question: str, vectors, chunks) -> str:
    relevant = retrieve(question, vectors, chunks, top_k=3)
    context = "\n\n---\n\n".join(r["text"] for r in relevant)
    prompt = f"""Answer the question using only the context below.
If the answer is not in the context, say "I don't know."

Context:
{context}

Question: {question}"""
    msg = ac.messages.create(
        model="claude-opus-4-5", max_tokens=512,
        messages=[{"role":"user","content":prompt}]
    )
    return msg.content[0].text

answer = rag_answer("What is the return policy?", vectors, chunks)
print(answer)
Production tip: For larger document sets, use a proper vector database like Chroma, Pinecone, or Weaviate instead of NumPy arrays. They handle indexing, persistence, and fast approximate nearest-neighbour search at scale.

def ask(question):
    q_vec = embed([question])[0]
    scores = vectors @ q_vec  # dot product for similarity
    top3 = np.argsort(scores)[::-1][:3]
    context = " ".join(chunks[i] for i in top3)
    msg = ac.messages.create(
        model="claude-opus-4-5", max_tokens=300,
        messages=[{"role":"user","content":f"Context: {context}\n\nQ: {question}"}]
    )
    return msg.content[0].text

print(ask("When was Python created?"))

print(ask("What is Python used for?"))
