Async Python & Concurrency

Use asyncio and aiohttp to make multiple AI API calls concurrently — turning 60 seconds of sequential waits into 6.

35 min Intermediate Hands-on Lab 5 Quiz Questions

Why Async Matters for AI Apps

AI API calls are slow — a GPT-4 response can take 5–30 seconds. If your app calls 10 APIs sequentially, you wait 50–300 seconds. With async, you launch all 10 calls simultaneously and wait for the slowest one — reducing total time to that single call's duration.

The key insight: Async is not about running code in parallel threads. It's about not wasting time waiting. While one API call is in-flight, your program handles other work.

async / await Basics

import asyncio

# Define an async function
async def fetch_data(name: str) -> str:
    print(f"Starting {name}")
    await asyncio.sleep(1)  # simulates an API call
    print(f"Done {name}")
    return f"Result from {name}"

# Run it
result = asyncio.run(fetch_data("task1"))
print(result)

Running Tasks Concurrently

import asyncio, time

async def api_call(n: int) -> str:
    await asyncio.sleep(2)  # each call takes 2 seconds
    return f"Response {n}"

async def main():
    start = time.time()
    # Sequential: 3 * 2 = 6 seconds
    # results = [await api_call(i) for i in range(3)]
    
    # Concurrent: max(2,2,2) = 2 seconds
    results = await asyncio.gather(
        api_call(1), api_call(2), api_call(3)
    )
    print(f"Time: {time.time()-start:.1f}s")  # ~2.0s not 6.0s
    print(results)

asyncio.run(main())

Async with Real HTTP (aiohttp)

import asyncio, aiohttp

async def fetch(session, url):
    async with session.get(url) as resp:
        return await resp.json()

async def main():
    urls = [
        "https://api.github.com/users/python",
        "https://api.github.com/users/tensorflow",
        "https://api.github.com/users/pytorch",
    ]
    async with aiohttp.ClientSession() as session:
        tasks = [fetch(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
    for r in results:
        print(r["login"], r["public_repos"])

asyncio.run(main())

Async with the Anthropic SDK

import asyncio, anthropic, os

client = anthropic.AsyncAnthropic(api_key=os.environ["ANTHROPIC_API_KEY"])

async def summarise(text: str) -> str:
    msg = await client.messages.create(
        model="claude-opus-4-5", max_tokens=200,
        messages=[{"role":"user","content":f"Summarise in 1 sentence: {text}"}]
    )
    return msg.content[0].text

async def main():
    texts = ["Text about AI...", "Text about Python...", "Text about async..."]
    summaries = await asyncio.gather(*[summarise(t) for t in texts])
    for s in summaries: print(s)

asyncio.run(main())  # 3 calls in parallel!
Rule of thumb: Use async when your bottleneck is waiting (I/O — network, disk). Use multiprocessing when your bottleneck is computation (CPU-heavy math). Most AI app bottlenecks are I/O.

Sequential vs parallel API calls

import aiohttp, time

urls = [
    "https://api.github.com/users/python",
    "https://api.github.com/users/tensorflow",
    "https://api.github.com/users/keras-team",
    "https://api.github.com/users/huggingface",
]

async def fetch(session, url):
    async with session.get(url) as r:
        d = await r.json()
        return d["login"], d["public_repos"]

async def main():
    async with aiohttp.ClientSession() as session:
        start = time.time()
        results = await asyncio.gather(*[fetch(session, u) for u in urls])
        print(f"4 requests in {time.time()-start:.2f}s")
        for login, repos in results:
            print(f"{login}: {repos} repos")

await main()

Error handling in async

async def safe_fetch(session, url):
    try:
        async with session.get(url, timeout=aiohttp.ClientTimeout(total=5)) as r:
            r.raise_for_status()
            return await r.json()
    except Exception as e:
        return {"error": str(e), "url": url}

async def main():
    test_urls = urls + ["https://api.github.com/nonexistent/fake/repo"]
    async with aiohttp.ClientSession() as session:
        results = await asyncio.gather(*[safe_fetch(session, u) for u in test_urls])
    for r in results:
        if "error" in r:
            print(f"Failed: {r}")
        else:
            print(f"OK: {r.get('login', r.get('message', '??'))}")

await main()
