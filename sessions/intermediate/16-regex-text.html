Text Processing Is the Foundation of NLP

Before data reaches an LLM, it must be cleaned. Raw text from the web, PDFs, emails, and databases is messy: inconsistent whitespace, HTML tags, phone numbers, URLs, special characters. Regular expressions (regex) and Python's string tools are your first line of defence.

Python String Methods (the Basics)

text = "  Hello, World! This is Python 3.  "

# Cleaning
text.strip()          # remove leading/trailing whitespace
text.lower()          # "  hello, world! ..."
text.upper()          # uppercase
text.replace("!", ".")  # replace characters

# Splitting and joining
words = text.split()           # split on whitespace
sentences = text.split(".")    # split on period
", ".join(["a","b","c"])       # "a, b, c"

# Checking
text.startswith("Hello")  # False (has leading space)
text.strip().startswith("Hello")  # True
"Python" in text  # True
text.isdigit()  # False
"42".isdigit()  # True

Regular Expressions

import re

text = "Contact us at support@example.com or sales@company.org. Call 555-123-4567."

# Find all email addresses
emails = re.findall(r'[\w.-]+@[\w.-]+\.[a-z]{2,}', text)
print(emails)  # ['support@example.com', 'sales@company.org']

# Find phone numbers
phones = re.findall(r'\d{3}-\d{3}-\d{4}', text)
print(phones)  # ['555-123-4567']

# Replace (clean HTML tags)
html = "
Hello World
"
clean = re.sub(r'<[^>]+>', '', html)
print(clean)  # "Hello World"

# Check if string matches a pattern
if re.match(r'^\d{4}-\d{2}-\d{2}$', "2026-05-15"):
    print("Valid date format")

Text Cleaning Pipeline for LLMs

import re

def clean_for_llm(text: str) -> str:
    """Clean raw text before sending to an LLM."""
    # Remove HTML tags
    text = re.sub(r'<[^>]+>', ' ', text)
    # Remove URLs
    text = re.sub(r'https?://\S+', ' ', text)
    # Remove excessive whitespace
    text = re.sub(r'\s+', ' ', text)
    # Remove special characters (keep punctuation)
    text = re.sub(r'[^\w\s.,!?;:\'"-]', ' ', text)
    return text.strip()

raw = "Visit https://example.com! Get \t  started   today!!!"
print(clean_for_llm(raw))
# "Visit  Get  started   today!!!"

Tokenisation and Chunking

# Rough token counting (OpenAI: ~4 chars per token)
def estimate_tokens(text): return len(text) // 4

# Chunk long text for LLM context windows
def chunk_text(text, chunk_size=2000, overlap=200):
    chunks = []
    start = 0
    while start < len(text):
        end = start + chunk_size
        chunks.append(text[start:end])
        start += chunk_size - overlap  # overlap for context
    return chunks

long_text = "word " * 5000  # ~5000 words
chunks = chunk_text(long_text)
print(f"{len(chunks)} chunks, first chunk: {len(chunks[0])} chars")

Extract emails and phones from text

import re

raw = """
Dear team,
Please contact Alice at alice@company.com or Bob at b.jones@startup.io.
For urgent matters: call 415-555-0101 or 800-123-9876.
Details at https://internal.company.com/policy.html
"""

emails = re.findall(r'[\w.+-]+@[\w-]+\.[a-z]{2,}', raw)
phones = re.findall(r'\d{3}-\d{3}-\d{4}', raw)
urls   = re.findall(r'https?://\S+', raw)

print("Emails:", emails)
print("Phones:", phones)
print("URLs:", urls)
2

Clean the text

def clean(text):
    text = re.sub(r'https?://\S+', '', text)   # remove URLs
    text = re.sub(r'<[^>]+>', ' ', text)       # remove HTML
    text = re.sub(r'\s+', ' ', text)           # collapse whitespace
    return text.strip()

print(clean(raw))

Validate structured data with regex

def validate_date(s):
    return bool(re.match(r'^\d{4}-\d{2}-\d{2}$', s))

tests = ["2026-05-15", "05/15/2026", "2026-13-45", "2026-01-01"]
for t in tests:
    print(f"{t:15s} valid={validate_date(t)}")
