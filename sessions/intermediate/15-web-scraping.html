Web Scraping for AI Training Data

AI models are only as good as their training data. When public datasets don't exist, you collect your own. Web scraping lets you turn any website into a structured dataset. The tool of choice is BeautifulSoup for parsing HTML and requests for fetching pages.

Ethics first: Always check a site's robots.txt before scraping. Respect rate limits. Never scrape login-protected content or personal data without permission. Many sites prohibit scraping in their terms of service.

Extracting Structured Data

data = []

Following Pagination

import time

all_quotes = []
base_url = "https://quotes.toscrape.com/page/{}"

for page in range(1, 6):  # pages 1-5
    resp = requests.get(base_url.format(page))
    soup = BeautifulSoup(resp.text, "lxml")
    quotes = soup.find_all("div", class_="quote")
    if not quotes: break
    for q in quotes:
        all_quotes.append({
            "text": q.find("span", class_="text").text.strip(),
            "author": q.find("small", class_="author").text.strip()
        })
    time.sleep(1)  # be polite -- 1 second between requests
    print(f"Page {page}: {len(all_quotes)} total")

Scrape 5 pages with pagination

import time, pandas as pd

all_quotes = []
for page in range(1, 6):
    resp = requests.get(f"https://quotes.toscrape.com/page/{page}")
    soup = BeautifulSoup(resp.text, "lxml")
    for q in soup.find_all("div", class_="quote"):
        all_quotes.append({
            "quote": q.find("span", class_="text").text.strip('\u201c\u201d'),
            "author": q.find("small", class_="author").text,
            "tags": ", ".join(t.text for t in q.find_all("a", class_="tag"))
        })
    time.sleep(0.5)

df = pd.DataFrame(all_quotes)
print(df.shape)

Feed quotes to an AI

import anthropic, os
client = anthropic.Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY",""))

# Pick 5 quotes and ask AI to identify themes
sample = df.sample(5)["quote"].tolist()
quote_block = "\n".join(f"- {q}" for q in sample)

msg = client.messages.create(
    model="claude-opus-4-5", max_tokens=300,
    messages=[{"role":"user","content":f"Identify the 3 main themes across these quotes:\n{quote_block}"}]
)
print(msg.content[0].text)
